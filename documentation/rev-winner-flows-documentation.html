4. Payment Verification Flow (Razorpay)

Security: All payments verified using HMAC SHA256 signature validation
Provider: Razorpay Payment Gateway

Verification Process:

1 Razorpay payment handler receives credentials
Data from Razorpay:
• razorpay_order_id
• razorpay_payment_id
• razorpay_signature (HMAC SHA256 hash)
2 Frontend sends verification request
Endpoint: /api/train-me/verify-payment OR /api/session-minutes/verify-payment
Method: POST
Headers: Authorization: Bearer {accessToken}
Body: Payment credentials from step 1
3 Backend retrieves Razorpay secret key
Environment: RAZORPAY_KEY_SECRET
Security: Never exposed to frontend
4 Generate expected signature
Algorithm: HMAC SHA256
Message: razorpay_order_id + "|" + razorpay_payment_id
Secret: RAZORPAY_KEY_SECRET
Output: Expected signature hash
5 Compare signatures
Expected signature (from step 4) === razorpay_signature (from Razorpay)
Uses constant-time comparison to prevent timing attacks
6a If signatures match: Payment VERIFIED ✅
• Update database (activate subscription or add minutes)
• Return success response to frontend
• Frontend shows success toast and refreshes data
6b If signatures don't match: Payment REJECTED ❌
• Log security event
• Return 400 error to frontend
• Frontend shows error toast
• No database changes made
⚠️ Security Considerations:
• Signature validation prevents payment tampering
• Order ID uniqueness prevents duplicate processing
• Authentication required for all verification requests
• Server-side secret never exposed to client
• Failed verifications are logged for security monitoring

server/routes-auth.ts - Registration Integration
app.post("/api/auth/register", async (req: Request, res: Response) => {
  try {
    const data = registerUserSchema.parse(req.body);
    
    // Validate email and username uniqueness
    const existingEmail = await authStorage.getUserByEmail(data.email);
    if (existingEmail) {
      return res.status(400).json({ message: "Email already registered" });
    }
    
    const existingUsername = await authStorage.getUserByUsername(data.username);
    if (existingUsername) {
      return res.status(400).json({ message: "Username already taken" });
    }
    
    // Create pending user
    const user = await authStorage.createUser(data);
    
    // Generate and send OTP
    const otpCode = Math.floor(100000 + Math.random() * 900000).toString();
    const expiresAt = new Date(Date.now() + 10 * 60 * 1000); // 10 minutes
    
    await authStorage.createOTP(data.email, otpCode, expiresAt);
    await sendOTPEmail(data.email, otpCode, data.firstName);
    
    // Send registration notification to sales team (non-blocking)
    sendNewRegistrationNotification({
      firstName: data.firstName,
      lastName: data.lastName,
      email: data.email,
      username: data.username,
      company: data.organization,
    }).catch(err => console.error('Failed to send registration notification:', err));
    
    // Log audit
    await authStorage.createAuditLog({
      actorId: user.id,
      action: 'user.registered',
      targetType: 'user',
      targetId: user.id,
      metadata: { email: data.email, username: data.username },
      ipAddress: req.ip,
      userAgent: req.get('user-agent'),
    });
    
    res.status(201).json({
      message: "Registration successful. Please check your email for verification code.",
      userId: user.id,
      email: user.email,
    });
  } catch (error) {
    if (error instanceof z.ZodError) {
      return res.status(400).json({ message: "Validation error", errors: error.errors });
    }
    console.error("Registration error:", error);
    res.status(500).json({ message: "Registration failed" });
  }
});

6.2 Train Me Add-on Purchase

client/src/pages/subscribe.tsx - purchaseTrainMe()
async function purchaseTrainMe() {
  try {
    setIsPurchasingTrainMe(true);
    
    // Validate Razorpay loaded
    if (typeof window.Razorpay === 'undefined') {
      toast({
        title: "Payment Gateway Unavailable",
        description: "Please refresh the page and try again.",
        variant: "destructive",
      });
      setIsPurchasingTrainMe(false);
      return;
    }
    
    // Check authentication
    const accessToken = localStorage.getItem("accessToken");
    if (!accessToken) {
      toast({
        title: "Authentication Required",
        description: "Please log in to purchase Train Me.",
        variant: "destructive",
      });
      setLocation("/login");
      setIsPurchasingTrainMe(false);
      return;
    }

    // Create Razorpay order
    const orderResponse = await fetch("/api/train-me/create-order", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${accessToken}`,
      },
    });

    if (!orderResponse.ok) {
      const error = await orderResponse.json();
      throw new Error(error.message || "Failed to create order");
    }

    const { orderId, amount, currency, keyId } = await orderResponse.json();

    // Fetch user details for prefill
    const userResponse = await fetch("/api/auth/me", {
      headers: { "Authorization": `Bearer ${accessToken}` },
    });
    const userData = await userResponse.json();

    // Test mode detection
    const isTestMode = keyId.startsWith('rzp_test_');
    if (isTestMode) {
      toast({
        title: "Test Mode Active",
        description: "Using Razorpay test mode. Use test card: 4111 1111 1111 1111",
        duration: 10000,
      });
    }

    // Razorpay configuration
    const options = {
      key: keyId, // API key from backend (NEVER hardcoded)
      amount: amount,
      currency: currency,
      name: "Rev Winner",
      description: "Train Me Add-on (30 days)",
      order_id: orderId,
      handler: async function (response: any) {
        try {
          // Verify payment on backend
          const verifyResponse = await fetch("/api/train-me/verify-payment", {
            method: "POST",
            headers: {
              "Content-Type": "application/json",
              "Authorization": `Bearer ${accessToken}`,
            },
            body: JSON.stringify({
              razorpay_order_id: response.razorpay_order_id,
              razorpay_payment_id: response.razorpay_payment_id,
              razorpay_signature: response.razorpay_signature,
            }),
          });

          const verifyData = await verifyResponse.json();

          if (verifyResponse.ok) {
            queryClient.invalidateQueries({ queryKey: ["/api/train-me/status"] });
            toast({
              title: "Train Me Activated!",
              description: "You now have 30 days of access to Train Me features.",
            });
          } else {
            toast({
              title: "Verification Failed",
              description: verifyData.message || "Payment verification failed.",
              variant: "destructive",
            });
          }
        } catch (error: any) {
          toast({
            title: "Verification Error",
            description: error.message || "Failed to verify payment.",
            variant: "destructive",
          });
        }
      },
      prefill: {
        name: userData.user?.firstName ? 
          `${userData.user.firstName} ${userData.user.lastName}` : "",
        email: userData.user?.email || "",
      },
      theme: { color: "#C026D3" }, // Rev Winner purple
      modal: {
        ondismiss: function() {
          toast({ 
            title: "Payment Cancelled", 
            description: "You cancelled the payment process." 
          });
        },
      },
    };

    const razorpay = new window.Razorpay(options);
    razorpay.open();
    setIsPurchasingTrainMe(false);
  } catch (error: any) {
    toast({
      title: "Purchase Failed",
      description: error.message || "Failed to initiate payment. Please try again.",
      variant: "destructive",
    });
    setIsPurchasingTrainMe(false);
  }
}
